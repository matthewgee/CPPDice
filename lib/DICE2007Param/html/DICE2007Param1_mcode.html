1% Creates a param struct with the default parameters from DICE-2007
  2% using values taken from the GAMS code at econ.yale.edu/~nordhaus/DICE2007_programs/
  3
  4function [param] = DICE2007Param() %#codegen
  5    
  6    % Set default values for model parameters
  7    tmax = 60;
  8    param.elasmu = 2.0; % Elasticity of marginal utility of consumption
  9    param.prstp = .015; % Initial rate of social time preference per year
 10    
 11    % the following were settings I made that were not in the GAMS code
 12    % most come from Bob Kopps matDICE
 13    param.tmax = 60; % Time periods, in decades (60 * 10 = 600 years)
 14    param.numScen = 1; % Number of scenarios to run
 15    param.savings = .2; % Savings rate (constant)
 16    param.miu_2005 = 0; % emission control rate (fraction of uncontrolled emissions)
 17    
 18    % Population and technology
 19    param.pop0 = 6514; % 2005 world population millions
 20    param.gpop0 = .35; % growth rate of population per decade
 21    param.popasym = 8600; % asymptotic population
 22    
 23    param.a0 = .02722; % Initial level of total factor productivity
 24    param.ga0 = .092; % Initial growth rate for technology per decade
 25    param.dela = .001; % Decline rate of technological change per decade
 26    
 27    param.dk = .100; % Depreciation rate on capital per year
 28    param.gama = .300; % Capital elasticity in production function
 29    
 30    param.q0 = 61.1; % 2005 world gross output trill 2005 US dollars
 31    param.k0 = 137.; % 2005 value capital trill 2005 US dollars
 32    
 33    % Emissions
 34    param.sig0 = .13418; % CO2-equivalent emissions-GNP ratio 2005 (effectively intensity)
 35    param.gsigma = -.0730; % Initial growth of sigma per decade
 36    param.dsig = .003; % Decline rate of decarbonization per decade
 37    param.dsig2 = .000; % Quadratic term in decarbonization
 38    param.eland0 = 11.000; % Carbon emissions from land 2005 (GtC per decade)
 39    param.e2005 = 84.1910; % Year 2005 Emissions
 40    
 41    % Carbon Cycle    
 42    param.mat2000 = 808.9; % Concentration in atmosphere 2005 (GtC)
 43    param.mu2000 = 1255; % Concentration in upper strata 2005 (GtC)
 44    param.ml2000 = 18365; % Concentration in lower strata 2005 (GtC)
 45    
 46    param.matPI = 278 * 2.13; % Preindustrial concentration in atmosphere 2005 (GtC)
 47    
 48    % Carbon cycle transition matrix
 49    % 1 = atmosphere
 50    % 2 = upper ocean
 51    % 3 = lower ocean
 52    % each carbon parameter has two values, the source and the destination
 53    % e.g. carbon12 is the carbon going FROM the atmosphere TO the upper ocean
 54    
 55    param.b11 = .810712;
 56    param.b12 = .189288;
 57    param.b21 = .097213;
 58    param.b22 = .852787;
 59    param.b23 = .05;
 60    param.b32 = .003119;
 61    param.b33 = .996881;
 62    
 63    % Climate model
 64    param.t2xco2 = 3; % Equilibrium temp impact of CO2 doubling (degrees C)
 65    param.fex0 = -.06; % Estimate of 2000 forcings of non-CO2 GHG
 66    param.fex1 = .30; % Estimate of 2100 forcings of non-CO2 GHG
 67    
 68    param.tocean0 = .0068; % 2000 lower strat. temp change (C) from 1900
 69    param.tatm0 = .7307; % 2000 atmospheric temp change (C) from 1900
 70    
 71    param.c1 = .220; % Climate-equation coefficient for upper level
 72    param.c3 = .300; % Transfer coefficient upper to lower stratum
 73    param.c4 = .050; % Transfer coefficient for lower level
 74    
 75    param.fco22x = 3.8; % Estimated forcings of equilibrium CO2 doubling
 76    
 77    % Climate damage parameters, calibrated for quadratic at 2.5 C for 2105
 78    param.a1 = 0.00000; % Damage intercept
 79    param.a2 = 0.0028388; % Damage quadratic term
 80    param.a3 = 2.00; % Damage exponent
 81    
 82    % Abatement cost
 83    param.expcost2 = 2.8; % Exponent of control cost function
 84    param.pback = 1.17; % Cost of backstop 2005, thousands of $ per tC 2005
 85    param.backrat = 2; % Ratio initial to final backstop cost
 86    param.gback = .05; % Initial cost decline backstop, % per decade
 87    param.limmiu = 1; % Upper limit on control rate
 88    
 89    % Participation
 90    param.partfract1 = 1; % Fraction of emissions under control regime 2005
 91    param.partfract2 = 1; % Fraction of emissions under control regime 2015
 92    param.partfract21 = 1; % Fraction of emissions under control regime 2205
 93    param.dpartfract = 0; % Decline rate of participation
 94    
 95    % Availability of fossil fuels
 96    param.fosslim = 6000; % Maximum cumulative extraction fossil fuels
 97    
 98    % Scaling and inessential parameters
 99    param.scale1 = 194; % Scaling coefficient in the objective function
100    param.scale2 = 381800; % Scaling coefficient in the objective function
101    
102    % Initialized initial values for endogenous variables
103    param.capital = 0;
104    param.capital = 0;
105    param.output = 0;
106
107    param.massAtmosphere = 0;
108    param.massUpper = 0;
109    param.massLower = 0;
110
111    param.tempAtmosphere = 0;
112    param.tempLower = 0;
113
114    param.lam = 0; %Initialize climate model parameter
115
116
117      % Set up the arrays for the exogenous variables
118
119      param.miu = zeros(tmax, 1); % Emissions control rate (zero for now)
120      
121      param.e2005cap = ones(tmax, 1); % Emissions cap as a percentage of year 2005 emissions
122
123      % Population
124      param.gfacpop = zeros(tmax, 1); % Growth factor population
125      
126      param.l = zeros(tmax, 1); % Labor (population)
127
128      % Productivity
129      param.ga = zeros(tmax, 1); % Growth rate of productivity from 0 to tmax
130      
131      param.al = zeros(tmax, 1); % level of total factor productivity
132
133      % Climate
134      param.gsig = zeros(tmax, 1); % Cumulative improvement of energy efficiency
135      
136      param.sigma = zeros(tmax, 1); % CO2-equivalent-emissions output ratio
137      
138      param.gcost1 = zeros(tmax, 1); % growth of cost factor
139      
140      param.etree = zeros(tmax, 1); % Emissions from deforestation
141
142      param.rr = zeros(tmax, 1); % Average utility social discount rate
143      
144      param.forcoth = zeros(tmax, 1); % Exogenous forcing for other greenhouse gases
145      
146      param.partfract = zeros(tmax, 1); % Fraction of emissions in control regime
147
148      param.lam = param.fco22x/ param.t2xco2; % Set climate model parameter
149      
150      %Set Initial Values for endogenous variables in period one
151      param.capital = param.k0;
152      
153      param.output = param.q0;
154
155      param.massAtmosphere = param.mat2000;
156      
157      param.massUpper = param.mu2000;
158      
159      param.massLower = param.ml2000;
160
161      param.tempAtmosphere = param.tatm0;
162      param.tempLower = param.tocean0;
163      
164      %Step through the time steps to set up endogenous variable arrays
165      % Consider alternative way of doing this
166      
167      for i = 1:param.tmax
168          param.miu(i) = param.miu_2005;
169          %param.miu(i) = 1;
170
171          param.gfacpop(i) = (exp(param.gpop0*(i-1))-1)/exp(param.gpop0*(i-1));
172          param.l(i) = param.pop0 * (1 - param.gfacpop(i)) + param.gfacpop(i) * param.popasym;
173
174
175          param.gsig(i) = param.gsigma * exp(-param.dsig * 10 * (i - 1) - param.dsig2 * 10 * ((i - 1)^2));
176          param.etree(i) = param.eland0 * (1 - 0.1)^(i-1);
177    param.rr(i) = 1 / ((1 + param.prstp)^(10 * (i-1)));
178
179          param.ga(i) = param.ga0 * exp(-param.dela * 10 * (i-1));
180          if i == 1
181              param.al(i) = param.a0;
182              param.sigma(i) = param.sig0;
183          else
184              param.al(i) = .95 * param.al(i-1) / (1 - param.ga(i-1));
185        %param.al(i) = param.a0;
186              param.sigma(i) = param.sigma(i-1)/(1-param.gsig(i));
187          end
188
189          if i < 12
190              param.forcoth(i) = param.fex0 + .1 * (param.fex1 - param.fex0) * (i-1);
191          elseif i >= 12
192              param.forcoth(i) = param.fex0 + .36;
193          end
194
195          if i == 1
196              param.partfract(i) = param.partfract1;
197          elseif i < 25
198              param.partfract(i) = param.partfract21 + (param.partfract2 - param.partfract21) * exp(-param.dpartfract * (i-2));
199          else
200              param.partfract(i) = param.partfract21;
201          end
202
203          param.gcost1(i) = (param.pback * param.sigma(i) / param.expcost2) * (param.backrat - 1 + exp(-param.gback * (i-1))/param.backrat);
204
205      end
206    
207end
