 98function [vars] = DICE2007Utility(vars, param, t)
 99    vars.investment(t) = (param.savings) * vars.output(t);
100    % A.7
101    vars.consumption(t) = (1 - param.savings) * vars.output(t);
102    % A.8
103    vars.consumptionpercapita(t) = 1000* vars.consumption(t) / param.l(t);
104    % A.3
105    vars.utility(t) = param.l(t) * ((vars.consumptionpercapita(t))^(1-param.elasmu) / (1-param.elasmu));
106    vars.utilitydiscounted(t) = vars.utility(t) * param.rr(t);
107end
