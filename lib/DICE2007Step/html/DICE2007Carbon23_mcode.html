 63function [vars] = DICE2007Carbon(vars, param, t)
 64  if t~=1
 65    % A.13
 66    vars.massAtmosphere(t) = vars.emissionsTotal(t-1) + param.b11 * vars.massAtmosphere(t-1) + param.b21*vars.massUpper(t-1);
 67    % A.14
 68    vars.massUpper(t) = param.b12*vars.massAtmosphere(t-1) + param.b22*vars.massUpper(t-1) + param.b32*vars.massLower(t-1);
 69    % A.15
 70    vars.massLower(t) = param.b23*vars.massUpper(t-1) + param.b33*vars.massLower(t-1);
