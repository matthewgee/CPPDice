 85function [vars] = DICE2007Damages(vars, param, t)
 86    % miu
 87    vars.participation(t) = param.partfract(t);
 88    % A.19
 89    vars.participationMarkup(t) = vars.participation(t)^(1-param.expcost2);
 90    % A.5
 91    vars.damage(t) = 1/( 1 + param.a1 * vars.tempAtmosphere(t) + param.a2 * vars.tempAtmosphere(t)^param.a3);
 92    % A.6
 93    vars.abatement(t) = vars.participationMarkup(t) * param.gcost1(t) * (param.miu(t) ^ param.expcost2);
 94    % A.4
 95    vars.output(t) = vars.grossoutput(t)*vars.damage(t)*(1-vars.abatement(t));
 96end
